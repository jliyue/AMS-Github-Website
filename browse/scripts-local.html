<script>


function displayBrowseTable(data, selector) {

let element = document.querySelector(selector || "#list");

if (!element) {

return;

}


let headings = ["FILE_ID", "file_name", "FOLDER_PROVINCE", "relative_path", "raw_github_url"];


element.innerHTML = `

<p><b>${data.length}</b> files loaded from metadata.</p>

<table class="browse">

<thead>${makeTableHeader(["ID", "File name", "Province", "Path", "File"])}</thead>

<tbody>${makeTableBody(data)}</tbody>

</table>

`;

}


function makeTableHeader(headings) {

return `<tr><th>${headings.join("</th><th>")}</th></tr>`;

}


function makeTableBody(data) {

return data.map(function(entry) {

let id = entry.FILE_ID || "";

let fileName = entry.file_name || "";

let province = entry.FOLDER_PROVINCE || "";

let path = entry.relative_path || "";

let rawUrl = entry.raw_github_url || "";


return `

<tr>

<td>${id}</td>

<td>${fileName}</td>

<td>${province}</td>

<td>${path}</td>

<td><a target="_blank" href="${rawUrl}">raw file</a></td>

</tr>

`;

}).join("");

}


</script>
